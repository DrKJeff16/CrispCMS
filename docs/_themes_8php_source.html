112  public static function installTranslations($ThemeName, $ThemeMetadata) {
113  if (!\is_object($ThemeMetadata) && !isset($ThemeMetadata->hookFile)) {
114  return false;
115  }
116 
117  $_processed = [];
118  if (defined("CRISP_CLI")) {
119  echo "----------" . PHP_EOL;
120  echo "Installing translations for theme $ThemeName" . PHP_EOL;
121  echo "----------" . PHP_EOL;
122  }
123 
124  if (isset($ThemeMetadata->onInstall->createTranslationKeys) && is_string($ThemeMetadata->onInstall->createTranslationKeys)) {
125

128

130  foreach ($files as $File) {

133  echo "----------" . PHP_EOL;
134  echo "Installing language " . substr(basename($File), 0, -5) . PHP_EOL;
135  echo "----------" . PHP_EOL;
136  }

138  continue;
139  }
140  $Language = \crisp\api\lists\Languages::getLanguageByCode(substr(basename($File), 0, -5));
141 
142  if (!$Language) {
143  continue;
144  }
145 
146  foreach (json_decode(file_get_contents($File), true) as $Key => $Value) {
147  try {
148 
149  if ($Language->newTranslation($Key, $Value, substr(basename($File), 0, -5))) {
150  $_processed[] = $Key;
151  if (defined("CRISP_CLI")) {
152  echo "Installing key $Key" . PHP_EOL;
153  }
154  }
155  } catch (\PDOException $ex) {
156  continue 2;
157  }
158  }
159 
160  if (defined("CRISP_CLI")) {
161  echo "Installed/Updated " . count($_processed) . " keys" . PHP_EOL;
162  $_processed = [];
163  }
164  }
165  }
166  return true;
167  }
168  if (isset($ThemeMetadata->onInstall->createTranslationKeys) && \is_object($ThemeMetadata->onInstall->createTranslationKeys)) {
169  foreach ($ThemeMetadata->onInstall->createTranslationKeys as $Key => $Value) {
170 
171  try {
172  $Language = \crisp\api\lists\Languages::getLanguageByCode($Key);
173 
174  if (!$Language) {
175  continue;
176  }
177 
178  foreach ($Value as $KeyTranslation => $ValueTranslation) {
179  $Language->newTranslation($KeyTranslation, $ValueTranslation, $Key);
180  }
181  } catch (\PDOException $ex) {
182  continue;
183  }
184  }
185  return true;
186  }
187  return false;
