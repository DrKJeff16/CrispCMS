123  public function migrate(string $Dir = __DIR__ . "/../", string $Plugin = null) {
124  if (defined("CRISP_CLI")) {
125  echo "Starting Migration..." . PHP_EOL;
126  }
127  if (!file_exists("$Dir/migrations/")) {
128  if (defined("CRISP_CLI")) {
129  echo "No migrations needed!" . PHP_EOL;
130  }
131  return;
132  }
133  $files = glob("$Dir/migrations/*.{php}", GLOB_BRACE);
134 
135  natsort($files);
136 
137  foreach ($files as $file) {
138  if (basename($file) == "template.php") {
139  continue;
140  }
141 
142  $MigrationName = substr(basename($file), 0, -4);
143 
144  if ($this->isMigrated($MigrationName)) {
145  if (defined("CRISP_CLI")) {
146  echo "$MigrationName is already migrated, skipping!" . PHP_EOL;
147  }
148  continue;
149  }
150 
151  $Class = "\crisp\migrations\\" . explode("_", $MigrationName)[1];
152 
153  include $file;
154 
155  $Migration = new $Class();
156 
157  if ($Migration->run()) {
158 
159  $statement = $this->Database->prepare("INSERT INTO schema_migration (file, plugin) VALUES (:file, :plugin)");
160 
161  $statement->execute(array(":file" => $MigrationName, ":plugin" => $Plugin));
162 
163  if (defined("CRISP_CLI")) {
164  echo "Migrated $MigrationName" . PHP_EOL;
165  }
166  } else {
167 
168  if (defined("CRISP_CLI")) {
169  echo "Failed to migrate $MigrationName" . PHP_EOL;
170  }
171  }
172  }
