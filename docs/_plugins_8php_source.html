627  public static function integrityCheck($PluginName, $PluginMetadata) {
628  $PluginFolder = \crisp\api\Config::get("plugin_dir");
629  $parsedConfigs = array();
630  $failedConfigs = array();
631  $failedChecks = array();
632  $parsedChecks = array();
633  $parsedFiles = array();
634  $failedFiles = array();
635  $integrity = true;
636  if (isset($PluginMetadata->onInstall->createKVStorageItems) && \is_object($PluginMetadata->onInstall->createKVStorageItems)) {
637  foreach ($PluginMetadata->onInstall->createKVStorageItems as $Key => $Value) {
638  if (is_array($Value) || \is_object($Value)) {
639  $Value = \serialize($Value);
640  }
641  try {
642  $KeyExists = \crisp\api\Config::exists("plugin_" . $PluginName . "_$Key");
643  $parsedConfigs[$Key] = $Value;
644  if (!$KeyExists) {
645  $failedConfigs[$Key] = $Value;
646  }
647  } catch (\PDOException $ex) {
648  $failedConfigs[$Key] = $Value;
649  }
650  }
651  $parsedChecks[] = "KVStorageItems";
652  if (count($failedConfigs) > 0) {
653  $integrity = false;
654  $failedChecks[] = "KVStorageItems";
655  }
656  }
657 
658 
659 
660  if (!self::isInstalled($PluginName)) {
661  $integrity = false;
662  $failedChecks[] = "isInstalled";
663  } else {
664  $parsedChecks[] = "isInstalled";
665  }
666 
667  $PluginPath = \realpath(__DIR__ . "/../../../../$PluginFolder" . DIRECTORY_SEPARATOR . "$PluginName") . DIRECTORY_SEPARATOR;
668  if (\file_exists($PluginPath . "sha256sum.txt")) {
669 
670  foreach (file($PluginPath . "sha256sum.txt") as $line) {
671  if (!ctype_space($line)) {
672  $line = explode(" ", $line);
673  $Hash = $line[0];
674  $File = trim($line[1]);
675  $HashFile = \hash_file("sha256", $PluginPath . $File);
676  $parsedFiles[$File] = $HashFile;
677  if ($HashFile != $Hash) {
678  $failedFiles[$File] = $HashFile;
679  }
680  }
681  }
682  if (count($failedFiles) > 0) {
683  $integrity = false;
684  $failedChecks[] = "hash_file";
685  }
686  $parsedChecks[] = "hash_file";
687  }
688 
689  return array(
690  "integrity" => $integrity,
691  "parsedConfigs" => $parsedConfigs,
692  "failedConfigs" => $failedConfigs,
693  "parsedFiles" => $parsedFiles,
694  "failedFiles" => $failedFiles,
695  "isInstalled" => self::isInstalled($PluginName),
696  );
697  }
