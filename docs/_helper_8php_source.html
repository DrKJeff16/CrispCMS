288  public static function isSerialized($data, $strict = true) {
289  // if it isn't a string, it isn't serialized.
290  if (!is_string($data)) {
291  return false;
292  }
293  $data = trim($data);
294  if ('N;' == $data) {
295  return true;
296  }
297  if (strlen($data) < 4) {
298  return false;
299  }
300  if (':' !== $data[1]) {
301  return false;
302  }
303  if ($strict) {
304  $lastc = substr($data, -1);
305  if (';' !== $lastc && '}' !== $lastc) {
306  return false;
307  }
308  } else {
309  $semicolon = strpos($data, ';');
310  $brace = strpos($data, '}');
311  // Either ; or } must exist.
312  if (false === $semicolon && false === $brace)
313  return false;
314  // But neither must be in the first X characters.
315  if (false !== $semicolon && $semicolon < 3)
316  return false;
317  if (false !== $brace && $brace < 4)
318  return false;
319  }
320  $token = $data[0];
321  switch ($token) {
322  case 's' :
323  if ($strict) {
324  if ('"' !== substr($data, -2, 1)) {
325  return false;
326  }
327  } elseif (false === strpos($data, '"')) {
328  return false;
329  }
330  // or else fall through
331  case 'a' :
332  case 'O' :
333  return (bool) preg_match("/^{$token}:[0-9]+:/s", $data);
334  case 'b' :
335  case 'i' :
336  case 'd' :
337  $end = $strict ? '$' : '';
338  return (bool) preg_match("/^{$token}:[0-9.E-]+;$end/", $data);
339  }
340  return false;
341  }
